public class FOR {
    public static void main(String[] args) {
        int n, sum = 0;
        for (int i =1; i<=5; i++) {
            n = (int)(Math.random() * 100);
            sum += n;
            System.out.printf("จำนวนที่ %d คือ %d\n", i, n);
        }
        System.out.println("รวมทั้งหมด:"+sum);
    }
}
